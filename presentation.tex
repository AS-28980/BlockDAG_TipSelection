\section{Blockchain Fundamentals}
% ==========================================================

\begin{frame}{What is a Blockchain?}
  \begin{columns}[T]
    \column{0.55\textwidth}
    A \textbf{blockchain} is a distributed, append-only ledger in which transactions are grouped into blocks chained together by cryptographic hashes.

    \bigskip
    \textbf{Key properties:}
    \begin{itemize}
      \item Decentralization --- no single controlling authority
      \item Immutability --- confirmed data cannot be altered
      \item Consensus --- PoW / PoS to agree on one chain
      \item Transparency --- every participant audits the full history
    \end{itemize}

    \column{0.42\textwidth}
    \textbf{Main applications:}
    \begin{itemize}
      \item Cryptocurrencies (Bitcoin, Ethereum)
      \item Supply-chain tracking \& logistics
      \item Smart contracts \& DeFi
      \item Sharing data in a trustworthy manner
      \item IoT device orchestration
    \end{itemize}
  \end{columns}
\end{frame}

% ----------------------------------------------------------
\begin{frame}{Problems with Blockchains}
  \begin{columns}[T]
    \column{0.48\textwidth}
    \textbf{Performance}
    \begin{itemize}
      \item \textbf{Low throughput:} Bitcoin $\approx 7$ tx/s, Ethereum $\approx 15$ tx/s.  Visa averages $\approx 1{,}700$ tx/s.
      \item \textbf{High latency:} Bitcoin block time $\approx 10$ min; finality can take an hour.
      \item \textbf{Scalability ceiling:} sequential block production limits concurrency fundamentally.
    \end{itemize}

    \column{0.48\textwidth}
    \textbf{Economics \& Sustainability}
    \begin{itemize}
      \item \textbf{Energy:} PoW mining rivals the electricity usage of small countries.
      \item \textbf{Fees:} miners demand fees, making micro-transactions uneconomical.
      \item \textbf{Centralization risk:} mining pools concentrate hash-power; economies of scale undermine decentralization.
    \end{itemize}
  \end{columns}

  \bigskip
  \centering
  \textit{These limitations motivate the search for alternative ledger architectures.}
\end{frame}

% ==========================================================
\section{BlockDAGs}
% ==========================================================

\begin{frame}{Introducing BlockDAGs}
  A \textbf{Directed Acyclic Graph (DAG)}-based ledger replaces the single chain: each new transaction can reference \emph{multiple} previous ones simultaneously.

  \bigskip
  \textbf{Advantages over blockchains:}
  \begin{itemize}
    \item \textbf{Parallel processing} --- many transactions issued concurrently, no block-slot bottleneck.
    \item \textbf{Scalable throughput} --- throughput grows with the number of active participants.
    \item \textbf{No dedicated miners} --- every transaction validates previous ones (self-serving consensus).
    \item \textbf{Fee-free potential} --- PoW is lightweight and per-transaction; no block reward necessary.
  \end{itemize}

  \bigskip
  Particularly attractive for \textbf{IoT}, smart-city applications, and high-frequency micro-trading where low fees and low latency are essential.
\end{frame}

% ----------------------------------------------------------
\begin{frame}{BlockDAGs vs.\ Blockchains}
  \centering
  \begin{tabular}{@{} l c c @{}}
    \toprule
    & \textbf{Blockchain} & \textbf{BlockDAG} \\
    \midrule
    Structure        & Linear chain           & Directed acyclic graph \\
    Concurrency      & 1 block per slot       & Many txs in parallel \\
    Throughput       & $\sim$7--15 tx/s       & Scales with users \\
    Micro-txs        & Impractical (fees)     & Near-zero cost \\
    Confirmation     & Minutes                & Seconds \\
    IoT suitability  & Heavy PoW, high fees   & Lightweight PoW, fee-free \\
    \bottomrule
  \end{tabular}
\end{frame}

% ----------------------------------------------------------
\begin{frame}{BlockDAGs: Limitations}
  BlockDAGs are \emph{not} a complete replacement for blockchains:
  \begin{itemize}
    \item \textbf{Smart-contract complexity:} state-dependent execution needs a total order on transactions; DAGs make global ordering harder.
    \item \textbf{Probabilistic finality:} most DAG protocols provide weaker finality guarantees than PoS chains --- problematic for financial settlement.
    \item \textbf{Ecosystem maturity:} tooling, auditing frameworks, and developer libraries lag far behind Ethereum / Solidity.
    \item \textbf{Coordinator dependency:} IOTA historically relied on a centralized Coordinator; fully decentralized DAG consensus remains an open research problem.
    \item \textbf{Double-spend vulnerability:} without a carefully designed tip-selection rule, attackers can exploit the DAG structure.  The tension between security parameter $\alpha$ and liveness is non-trivial --- and is precisely what motivates this project.
  \end{itemize}
\end{frame}

% ==========================================================
\section{IOTA's Tangle}
% ==========================================================

\begin{frame}{The Tangle Architecture}
  The \textbf{Tangle}~\cite{popov2018} is IOTA's DAG-based DLT, designed for feeless IoT micro-transactions.

  \bigskip
  \begin{itemize}
    \item Each new transaction (``site'') selects and approves $m=2$ previous transactions.
    \item Unapproved transactions form the \textbf{tip set} --- the growing frontier of the DAG.
    \item Before being added, a transaction must complete a lightweight \textbf{Proof of Work} taking time~$h$.
    \item The \textbf{cumulative weight} of a site equals the number of transactions that approve it (directly or indirectly).
  \end{itemize}

  \bigskip
  \textbf{Central metric:} the tip count $L(t)$ at time $t$.
  \begin{itemize}
    \item $L(t)$ bounded $\Rightarrow$ all transactions are eventually approved.
    \item $L(t) \to \infty$ $\Rightarrow$ orphaned transactions accumulate.
  \end{itemize}
\end{frame}

\section{MCMC Tip Selection}
% ==========================================================

\begin{frame}{Tip Selection Algorithms}
  \begin{columns}[T]
    \column{0.48\textwidth}
    \textbf{Random Selection (RS)}
    \begin{itemize}
      \item Pick $m$ tips uniformly at random.
      \item[$+$] Simple; $L(t)$ stays bounded.
      \item[$-$] Vulnerable to double-spend attacks.
    \end{itemize}

    \bigskip
    \textbf{The core dilemma:}\\
    Security ($\alpha\!\uparrow$) vs.\ Liveness ($\alpha\!\downarrow$).

    \column{0.48\textwidth}
    \textbf{MCMC Selection}
    \begin{itemize}
      \item Start $m$ random walks at the genesis.
      \item At site $j$, jump to child $k$ with probability
        \[
          P(j \!\to\! k) \;\propto\; e^{-\alpha\,(\vartheta_j - \vartheta_k)}
        \]
        where $\vartheta_i$ is the cumulative weight.
      \item Walk terminates at a tip.
      \item[$+$] Resistant to double-spend.
      \item[$-$] High $\alpha$ $\Rightarrow$ old tips orphaned.
    \end{itemize}
  \end{columns}
\end{frame}

% ----------------------------------------------------------
\begin{frame}{MCMC Walk --- Step by Step}
  \begin{enumerate}
    \item \textbf{Initialise:} launch $m=2$ independent walkers at the genesis site.
    \item \textbf{Transition:} at site $j$ with children $\{k_1,\dots,k_d\}$, move to $k_i$ with probability
      \[
        P(j \to k_i)
        \;=\;
        \frac{\exp\!\bigl(-\alpha\,(\vartheta_j - \vartheta_{k_i})\bigr)}
             {\displaystyle\sum_{k \in \text{children}(j)}
              \exp\!\bigl(-\alpha\,(\vartheta_j - \vartheta_{k})\bigr)}
      \]
    \item \textbf{Terminate:} when the walker reaches a tip (a site with no children), select that tip for approval.
    \item \textbf{Attach:} new transaction creates edges to its $m$ selected tips and performs PoW (delay $h$).
  \end{enumerate}

  \bigskip
  \textbf{Effect of $\alpha$:}
  \begin{itemize}
    \item $\alpha \to 0$: transition probabilities become uniform $\Rightarrow$ equivalent to random selection.
    \item $\alpha \to \infty$: walker follows the single heaviest-weight path $\Rightarrow$ deterministic, greedy.
  \end{itemize}
\end{frame}

\section{Mathematics of Tip Selection}
% ==========================================================

\begin{frame}{Fluid Model --- Setup}
  Ferraro et al.~\cite{ferraro2020} derive a \textbf{mean-field PDE} for the Tangle in the fluid limit ($\lambda\!\to\!\infty$, Poisson arrivals at rate $\lambda$).

  \bigskip
  \textbf{Key quantities} (rescaled by $\lambda^{-1}$):
  \[
    l(t) = \lim_{\lambda\to\infty} \lambda^{-1} L(t),
    \qquad
    x(t) = \lim_{\lambda\to\infty} \lambda^{-1} X(t),
    \qquad
    w(t) = \lim_{\lambda\to\infty} \lambda^{-1} W(t)
  \]
  where $L(t)$ = tips, $X(t)$ = free tips, $W(t)$ = pending tips ($W \le mh$).

  \bigskip
  Age-density $l(t,s)$: number of tips at time $t$ with age $s$ (time since added to DAG).

  \bigskip
  \textbf{Selection model:} each selection $j$ uses weight function $g_j(s)$ with normalisation
  \[
    \zeta_j(t) = \int_0^{t-h} g_j(s)\,l(t,s)\,\mathrm{d}s
  \]
  \begin{itemize}
    \item Random: $g(s)=1$ (constant, \emph{not} integrable).
    \item MCMC-like: $g(s)$ decays with age (\emph{integrable} for large $\alpha$).
  \end{itemize}
\end{frame}

% ----------------------------------------------------------
\begin{frame}{Fluid Model --- The Delayed PDE System}
  \textbf{Evolution of free-tip density} ($s>0$):
  \[
    \boxed{
    \frac{\partial x}{\partial t}(t,s)
    + \frac{\partial x}{\partial s}(t,s)
    \;=\;
    -\,x(t,s)\;\sum_{j=1}^{m}\frac{g_j(s)}{\zeta_j(t)}
    }
  \]

  \textbf{Evolution of total tip density:}
  \[
    \boxed{
    \frac{\partial l}{\partial t}(t,s)
    + \frac{\partial l}{\partial s}(t,s)
    \;=\;
    -\,x(t\!-\!h,\;s\!-\!h)\;\sum_{j=1}^{m}\frac{g_j(s-h)}{\zeta_j(t-h)}
    \;\cdot\;\theta(s-h)
    }
  \]

  \medskip
  with boundary conditions $x(t,0)=l(t,0)=1$ and
  \[
    \zeta_j(t) = \int_0^{t-h} g_j(s)\,l(t,s)\,\mathrm{d}s.
  \]

  \medskip
  These are \textbf{delayed PDEs}: the density at time $t$ depends on the state at time $t-h$ (the PoW delay).
\end{frame}

% ----------------------------------------------------------
\begin{frame}{Fluid Model --- Key Theorems}
  \textbf{Theorem~2 (Integrable weights $\Rightarrow$ orphan persistence):}

  If all $g_j$ satisfy $\displaystyle\int_0^\infty g_j(s)\,\mathrm{d}s < \infty$, then any steady-state solution satisfies
  \[
    \int_0^\infty l(s)\,\mathrm{d}s \;=\; \infty.
  \]
  Interpretation: under pure MCMC with large $\alpha$, the weight $g(s)$ decays fast (integrable), so selection concentrates on recent tips.  Old tips are never picked $\Rightarrow$ orphan count diverges.

  \bigskip
  \textbf{Theorem~3 (Random selection $\Rightarrow$ bounded tips):}

  If at least two selections are uniform random ($g_1=g_2=1$), then
  \[
    l(t) \;\le\; C \quad\text{for all } t\ge 0.
  \]
  All transactions are eventually approved.  However, random selection is vulnerable to double-spend attacks.
\end{frame}

% ----------------------------------------------------------
\begin{frame}{The Hybrid Tip-Selection Algorithm}
  Ferraro et al.~\cite{ferraro2020} propose a \textbf{hybrid} algorithm that resolves the security--liveness tension:

  \bigskip
  \begin{enumerate}
    \item \textbf{Security step:} first selection uses MCMC with a \emph{high} $\alpha$.
      \begin{itemize}
        \item Ensures that dishonest tips on low-weight paths are very unlikely to be chosen.
        \item Protects against double-spend attacks.
      \end{itemize}

    \bigskip
    \item \textbf{Swipe step:} second selection uses random selection (or MCMC with a very \emph{low} $\alpha \approx 0$).
      \begin{itemize}
        \item Guarantees that old, forgotten tips are eventually picked.
        \item Keeps $L(t)$ bounded $\Rightarrow$ no orphans.
      \end{itemize}
  \end{enumerate}

  \bigskip
  \textbf{Result:}  the hybrid algorithm inherits the security of high-$\alpha$ MCMC \emph{and} the stability guarantee of random selection.
\end{frame}

\section{Our Project}
% ==========================================================

\begin{frame}{Project Description}
  We implemented and experimentally validated the Tangle tip-selection model from Ferraro et al.~\cite{ferraro2020} with \textbf{two independent systems}:

  \bigskip
  \begin{columns}[T]
    \column{0.47\textwidth}
    \texttt{\textbf{tangle-sim}} (benchmark)
    \begin{itemize}
      \item Single-process Python/asyncio simulator.
      \item Discrete-event engine; priority-queue scheduling.
      \item 66 runs, 13\,074 transactions.
      \item Campaign time: $\sim$35 min.
    \end{itemize}

    \column{0.47\textwidth}
    \texttt{\textbf{tangle-distributed}} (validation)
    \begin{itemize}
      \item One OS process per node; TCP sockets.
      \item Real gossip protocol over the network.
      \item 47 runs, 9\,224 transactions.
      \item Campaign time: $\sim$25 min.
    \end{itemize}
  \end{columns}

  \bigskip
  \centering
  \textbf{Total: 113 runs \quad$\vert$\quad 22\,298 transactions \quad$\vert$\quad $\sim$55 min wall-clock.}
\end{frame}

% ----------------------------------------------------------
\begin{frame}{How We Improved Upon the Paper}
  Ferraro et al.\ used a single-process agent-based simulation.  We extended the work in several directions:
  \begin{itemize}
    \item \textbf{True distributed implementation:} one OS process per node with TCP-socket gossip --- real network I/O, not simulated.
    \item \textbf{Multiple topologies:} full-mesh, ring, small-world, random-$k$, star (paper assumed full connectivity).
    \item \textbf{Realistic network model:} log-normal latency with configurable base (20--800\,ms) and jitter.
    \item \textbf{Adversarial testing:} one node issuing at $4\times$ the honest rate; the paper treats double-spend only theoretically.
    \item \textbf{Cross-validation:} two independent codebases producing statistically identical results --- strong evidence of correctness.
    \item \textbf{Broader sweeps:} seven parameter dimensions ($\alpha$, algorithm, $\lambda$, $h$, $n$, topology, latency) totalling 113 runs versus the paper's focused simulations.
  \end{itemize}
\end{frame}

\section{Validation and Conclusions}
% ==========================================================

\begin{frame}{Validating the Paper's Claims}
  \begin{enumerate}
    \item \textbf{High-$\alpha$ MCMC $\Rightarrow$ orphan persistence.}\\
      {\color{green!60!black}$\checkmark$ Confirmed.}
      Tip count grows ${\sim}8\times$ from $\alpha\!=\!0.001$ to $\alpha\!=\!2$ in both implementations.

    \bigskip
    \item \textbf{Hybrid keeps $L(t)$ bounded while preserving security.}\\
      {\color{green!60!black}$\checkmark$ Confirmed.}
      Hybrid $\bar L\!=\!4.3$ vs.\ MCMC-high $\bar L\!=\!19.2$; security step uses $\alpha\!=\!1$.

    \bigskip
    \item \textbf{Mean-field prediction $L \approx 2\lambda h$.}\\
      {\color{green!60!black}$\checkmark$ Confirmed.}
      $\bar L$ scales linearly with $\lambda$ and $h$ across all configurations.

    \bigskip
    \item \textbf{System is stable at equilibrium.}\\
      {\color{green!60!black}$\checkmark$ Confirmed.}
      60\,s run matches 25\,s baselines exactly.

    \bigskip
    \item {\color{orange}\textbf{New finding:}} \textbf{Network latency is the critical operational parameter.}\\
      Convergence drops to 42\% at 800\,ms.  Not analysed in the original paper.
  \end{enumerate}
\end{frame}
